47 {
48  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkRed, "SPIN", count);
49  if (!InstructionSet::waitPackage())
50  {
51  for (uint32_t pp = 0; pp <= count; ++pp)

53  GTS_PAUSE();

55  }

85 GTS_INLINE void pauseFor(uint32_t cycleCount)
86 {
87  if (!InstructionSet::waitPackage())
88  {
89  int32_t count = 0;
90 
91  uint64_t prev = GTS_RDTSC();
92  const uint64_t finish = prev + cycleCount;
93  do
94  {
95  spin(count);
96  count *= 2;
97 
98  uint64_t curr = GTS_RDTSC();
99  if (curr <= prev)
100  {
101  // Quit on context switch or overflow.
102  break;
103  }
104  prev = curr;
105 
106  } while (prev < finish);
107  }
108  else
109  {
110  tpause(WAITPKG_OPTIMIZATION_STATE, cycleCount);
111  }
112 }

125 template<BackoffGrowth GROWTH_RATE = BackoffGrowth::Geometric, bool YIELD = true>
126 class Backoff
127 {
128 public:
129 
130  //--------------------------------------------------------------------------
135  GTS_INLINE bool tick()
136  {
137  if(!YIELD || m_cyclesSoFar < GTS_CONTEXT_SWITCH_CYCLES)
138  {
139  uint64_t begin = GTS_RDTSC();
140  spin(m_count);
141  uint64_t end = GTS_RDTSC();
142 
143  if (end <= begin)
144  {
145  // Context switch or overflow occurred. Assume we are ready to yield.
146  m_cyclesSoFar = GTS_CONTEXT_SWITCH_CYCLES;
147  }
148  else
149  {
150  m_cyclesSoFar += end - begin;
151  }
152 
153  switch (GROWTH_RATE)
154  {
155  case BackoffGrowth::Arithmetic:
156  m_count += 1;
157  break;
158  case BackoffGrowth::Geometric:
159  m_count *= 2;
160  break;
161  }
162  return false;
163  }
164  else if(YIELD)
165  {
166  GTS_TRACE_SCOPED_ZONE_P0(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkMagenta, "YIELD");
167  ThisThread::yield();
168  }
169 
170  return true;
171  }
172 
173  //--------------------------------------------------------------------------
178  GTS_INLINE void reset()
179  {
180  m_count = 1;
181  m_cyclesSoFar = 0;
182  }
183 
184  //--------------------------------------------------------------------------
191  GTS_INLINE int32_t spinCount() const
192  {
193  return m_count;
194  }
