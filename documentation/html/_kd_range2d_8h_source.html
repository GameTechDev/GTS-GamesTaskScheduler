96  GTS_INLINE KdRange2d(

139  template<typename TSplitter>
140  GTS_INLINE void split(split_result& result, TSplitter const& splitter)
141  {
142  result.size = MAX_SPLITS;
143  result.ranges[0] = *this;
144  KdRange2d& r = result.ranges[0];
145 
146  // If xRange is relatively smaller than yRange,
147  if (m_xRange.size() * m_yRange.minSize() < m_yRange.size() * m_xRange.minSize())
148  {
149  // split yRange.
150  r.m_yRange.begin() = y_range_type::splitHelper(m_yRange, splitter);
151  r.m_yRange.end() = m_yRange.end();
152  m_yRange.end() = r.m_yRange.begin();
153  }
154  else
155  {
156  // Otherwise, split xRange.
157  r.m_xRange.begin() = x_range_type::splitHelper(m_xRange, splitter);
158  r.m_xRange.end() = m_xRange.end();
159  m_xRange.end() = r.m_xRange.begin();
160  }
161  }
162

164  static GTS_INLINE uint16_t adjustDivisor(uint16_t divisor, bool isStatic)
165  {
166  if (isStatic)
167  {

174  static GTS_INLINE uint16_t splitDivisor(uint16_t divisor, bool isStatic)
175  {
176  if (isStatic)
177  {
178  return divisor - 1;
179  }
180  return divisor / SPLIT_FACTOR;
181  }
