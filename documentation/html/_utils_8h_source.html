253 GTS_INLINE uintptr_t alignDownTo(uintptr_t ptr, size_t alignment)
254 {
255  uintptr_t mask = alignment - 1;
256  if(isPow2(alignment))
257  {
258  return ptr & ~mask;
259  }
260  else
261  {
262  return (ptr / alignment) * alignment;
263  }
264 }

271 GTS_INLINE uintptr_t alignUpTo(uintptr_t ptr, size_t alignment)
272 {
273  uintptr_t mask = alignment - 1;
274  if(isPow2(alignment))
275  {
276  return (ptr + mask) & ~mask;
277  }
278  else
279  {
280  return ((ptr + mask) * alignment) / alignment;
281  }
282 }

392 //------------------------------------------------------------------------------
396 GTS_INLINE uint32_t murmur_hash3(const void * key, uint32_t len, uint32_t seed = 12345)
397 {
398  // https://github.com/aappleby/smhasher
399 
400  // 'm' and 'r' are mixing constants generated off-line.
401  // They're not really 'magic', they just happen to work well.
402 
403  const uint32_t m = 0x5bd1e995;
404  const int r = 24;
405 
406  // Initialize the hash to a 'random' value
407 
408  uint32_t h = seed ^ len;
409 
410  // Mix 4 bytes at a time into the hash
411 
412  const unsigned char * data = (const unsigned char *)key;
413 
414  while (len >= 4)
415  {
416  uint32_t k = *(uint32_t*)data;
417 
418  k *= m;
419  k ^= k >> r;
420  k *= m;
421 
422  h *= m;
423  h ^= k;
424 
425  data += 4;
426  len -= 4;
427  }
428 
429  // Handle the last few bytes of the input array
430 
431  switch (len)
432  {
433  case 3: h ^= data[2] << 16;
434  case 2: h ^= data[1] << 8;
435  case 1: h ^= data[0];
436  h *= m;
437  };
438 
439  // Do a few final mixes of the hash to ensure the last few
440  // bytes are well-incorporated.
441 
442  h ^= h >> 13;
443  h *= m;
444  h ^= h >> 15;
445 
446  return h;
447 }

549 template<typename Value, typename TCompareFunc>
550 GTS_INLINE void insertionSort(Value* values, size_t numValues, TCompareFunc fcnComp)
551 {
552  for (size_t ii = 1; ii < numValues; ++ii)
553  {
554  Value v = values[ii];
555  size_t jj = 0;
556 
557  while (jj < ii && fcnComp(v, values[ii - jj - 1]))
558  {
559  values[ii - jj] = values[ii - jj - 1];
560  ++jj;
561  }
562 
563  values[ii - jj] = v;
564 
565  }
566 }
