100  class FairSpinMutex
101  {
102  public:
103 
104  //----------------------------------------------------------------------
105  GTS_INLINE FairSpinMutex() : m_nextTicket(0), m_currTicket(0) {}
106 
107  //----------------------------------------------------------------------
108  GTS_INLINE void lock()
109  {
110  // Get the next ticket.
111  uint16_t myTicket = m_nextTicket.fetch_add(1, std::memory_order_acquire);
112 
113  // Wait until my ticket is up.
114  while (m_currTicket.load(std::memory_order_acquire) != myTicket)
115  {
116  GTS_PAUSE();
117  }
118  }
119 
120  //----------------------------------------------------------------------
121  GTS_INLINE void unlock()
122  {
123  // Mark the next ticket as ready.
124  m_currTicket.fetch_add(1, std::memory_order_release);
125  }
126 
127  private:
128 
129  std::atomic<uint16_t> m_nextTicket;
130  std::atomic<uint16_t> m_currTicket;
131  };

178  GTS_INLINE void logf(CaptureMask::Type captureMask, const char* fmt, ...)
179  {
180  if(captureMask & m_captureMask)
181  {
182  Lock<FairSpinMutex> lock(m_accessorMutex);
183 
184  if (m_pos + 1 >= m_capacity)
185  {
186  _resize();
187  }
188 
189  // Write an event at this index
190  ConcurrentLoggerEvent* e = &m_pEventsStack[m_pos++];
191 
192  e->tid = std::this_thread::get_id();
193  e->isMessageBuffer = true;
194

196  va_start(args, fmt);
197  vsprintf_s(e->message.buff, fmt, args);
198  va_end(args);

284  // Resizes the events array.
285  GTS_INLINE void _resize()
286  {
287  // double the buffer capacity
288  size_t newCapacity = m_capacity * 2;
289  ConcurrentLoggerEvent* pNewStack = (ConcurrentLoggerEvent*)GTS_ALIGNED_MALLOC(sizeof(ConcurrentLoggerEvent) * newCapacity, GTS_NO_SHARING_CACHE_LINE_SIZE);
290  assert(pNewStack);
291 
292  for (size_t ii = 0; ii < m_pos; ++ii)
293  {
294  pNewStack[ii] = m_pEventsStack[ii];
295  }
296 
297  if (m_pQuiescentEventsStack)
298  {
299  GTS_ALIGNED_FREE(m_pQuiescentEventsStack);
300  }
301  m_pQuiescentEventsStack = m_pEventsStack;
302 
303  m_capacity = newCapacity;
304  m_pEventsStack = pNewStack;
305  }
