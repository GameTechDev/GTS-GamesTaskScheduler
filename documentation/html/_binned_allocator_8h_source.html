573  GTS_INLINE uint32_t calculateBin(uint32_t size)
574  {
575  GTS_ASSERT(size > 0);
576 
577  uint32_t bin = 0;
578 
579  if (size <= BIN_SIZE_CLASS_0)
580  {
581  bin = (size - 1) / GTS_MALLOC_ALIGNEMNT;
582  }
583  else if (size <= BIN_SIZE_CLASS_2)
584  {
585  uint32_t upperBound = nextPow2(size + 1);
586  uint32_t lowerBound = upperBound / 2;
587  uint32_t intervalSize = (upperBound - lowerBound) / BIN_DIVISOR;
588  uint32_t localBin = (uint32_t)alignUpTo(size, intervalSize);
589  uint32_t numClass0Bins = BIN_SIZE_CLASS_0 / GTS_MALLOC_ALIGNEMNT - 1;
590  uint32_t class0Overlap = log2i(BIN_SIZE_CLASS_0) * BIN_DIVISOR;
591  uint32_t intervalStep = (localBin - lowerBound) / intervalSize;
592 
593  bin = (log2i(size) * BIN_DIVISOR + intervalStep) - class0Overlap + numClass0Bins;
594  }
595  else if (size <= BIN_SIZE_CLASS_3)
596  {
597  bin = m_numBins - 1;
598  }
599  else
600  {
601  bin = BIN_IDX_OVERSIZED;
602  }
603 
604  return bin;
605  }
606 
612  GTS_INLINE uint32_t calculateBin(void* ptr)
613  {
614  GTS_ASSERT(ptr);
615  SlabHeader* pSlab = MemoryStore::toSlab(ptr);
616  return calculateBin(pSlab->blockSize);
617  }
618 
619 public:
620 
621  static constexpr uint32_t BIN_SIZE_CLASS_0 = 1024;
622  static constexpr uint32_t BIN_SIZE_CLASS_1 = 8 * 1024;
623  static constexpr uint32_t BIN_SIZE_CLASS_2 = 32 * 1024;
624  static constexpr uint32_t BIN_SIZE_CLASS_3 = 512 * 1024;
625  static constexpr uint32_t BIN_IDX_OVERSIZED = UINT32_MAX;
626  static constexpr uint32_t BIN_DIVISOR = 4;
627
