609  GTS_INLINE size_type calculateBin(size_type size)
610  {
611  GTS_ASSERT(size > 0);
612 
613  size_type bin = 0;
614 
615  if (size <= BIN_SIZE_CLASS_0)
616  {
617  bin = (size - 1) / GTS_MALLOC_ALIGNEMNT;
618  }
619  else if (size <= BIN_SIZE_CLASS_2)
620  {
621  size_type upperBound = nextPow2(size + 1);
622  size_type lowerBound = upperBound / 2;
623  size_type intervalSize = (upperBound - lowerBound) / BIN_DIVISOR;
624  size_type localBin = (size_type)alignUpTo(size, intervalSize);
625  size_type numClass0Bins = BIN_SIZE_CLASS_0 / GTS_MALLOC_ALIGNEMNT - 1;
626  size_type class0Overlap = log2i(BIN_SIZE_CLASS_0) * BIN_DIVISOR;
627  size_type intervalStep = (localBin - lowerBound) / intervalSize;
628 
629  bin = (log2i(size) * BIN_DIVISOR + intervalStep) - class0Overlap + numClass0Bins;
630  }
631 #if GTS_64BIT
632  else if (size <= BIN_SIZE_CLASS_3)
633  {
634  bin = m_numBins - 1;
635  }
636 #endif
637  else
638  {
639  bin = BIN_IDX_OVERSIZED;
640  }
641 
642  return bin;
643  }
644 
650  GTS_INLINE size_type calculateBin(void* ptr)
651  {
652  GTS_ASSERT(ptr);
653  SlabHeader* pSlab = MemoryStore::toSlab(ptr);
654  PageHeader* pPage = MemoryStore::toPage(pSlab, ptr);
655  return calculateBin(pPage->blockSize);
656  }
