568  GTS_INLINE size_type calculateBin(size_type size)
569  {
570  GTS_ASSERT(size > 0);
571 
572  size_type bin = 0;
573 
574  if (size <= BIN_SIZE_CLASS_0)
575  {
576  bin = (size - 1) / GTS_MALLOC_ALIGNEMNT;
577  }
578  else if (size <= BIN_SIZE_CLASS_2)
579  {
580  size_type upperBound = nextPow2(size + 1);
581  size_type lowerBound = upperBound / 2;
582  size_type intervalSize = (upperBound - lowerBound) / BIN_DIVISOR;
583  size_type localBin = (size_type)alignUpTo(size, intervalSize);
584  size_type numClass0Bins = BIN_SIZE_CLASS_0 / GTS_MALLOC_ALIGNEMNT - 1;
585  size_type class0Overlap = log2i(BIN_SIZE_CLASS_0) * BIN_DIVISOR;
586  size_type intervalStep = (localBin - lowerBound) / intervalSize;
587 
588  bin = (log2i(size) * BIN_DIVISOR + intervalStep) - class0Overlap + numClass0Bins;
589  }
590 #if GTS_64BIT
591  else if (size <= BIN_SIZE_CLASS_3)
592  {
593  bin = m_numBins - 1;
594  }
595 #endif
596  else
597  {
598  bin = BIN_IDX_OVERSIZED;
599  }
600 
601  return bin;
602  }
603 
609  GTS_INLINE size_type calculateBin(void* ptr)
610  {
611  GTS_ASSERT(ptr);
612  SlabHeader* pSlab = MemoryStore::toSlab(ptr);
613  PageHeader* pPage = MemoryStore::toPage(pSlab, ptr);
614  return calculateBin(pPage->blockSize);
615  }
