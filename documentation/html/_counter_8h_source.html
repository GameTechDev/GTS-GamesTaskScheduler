210  void print(SubIdType const& ownerId, const char* file)
211  {
212  static const size_t COUNT_WIDTH = 20;
213 
214  std::ofstream outFile;
215  outFile.open(file, std::ios_base::out);
216  outFile << std::setfill('_');
217 
218  //
219  // Collect all the counters by owner ID.
220 
221  std::unordered_map<SubIdType, std::unordered_map<gts::OwnedId, ThreadInfo::array_type, OwnedIdHash>> counterMapsByOwnedId;
222 
223  for (size_t iThread = 0; iThread < m_threadInfo.size(); ++iThread)
224  {
225  ThreadInfo* pInfo = m_threadInfo[iThread];
226  for (auto mapIter = pInfo->countsByOwnedId.begin(); mapIter != pInfo->countsByOwnedId.end(); ++mapIter)
227  {
228  auto& countersArray = counterMapsByOwnedId[mapIter->first.ownerId()][mapIter->first];
229 
230  // Accumulate the counts.
231  for(size_t ii = 0; ii < TCounters::COUNT; ++ii)
232  {
233  countersArray[ii] += mapIter->second[ii];
234  }
235  }
236  }
237 
238  //
239  // Print counters by owner Id.
240 
241  uint64_t totals[TCounters::COUNT] = {0};
242 
243  for (auto localMapIter = counterMapsByOwnedId.begin(); localMapIter != counterMapsByOwnedId.end(); ++localMapIter)
244  {
245  if (localMapIter->first != UNKNOWN_SUBID && localMapIter->first != ownerId)
246  {
247  continue;
248  }
249 
250  outFile << "=============\n";
251  outFile << "OWNER ID: " << localMapIter->first << "\n";
252  outFile << "=============\n";
253 
254  for (auto countersIter = localMapIter->second.begin(); countersIter != localMapIter->second.end(); ++countersIter)
255  {
256  outFile << "=============\n";
257  outFile << "Local ID: " << countersIter->first.localId() << "\n";
258  outFile << "=============\n";
259 
260  for (size_t iCounter = 0; iCounter < TCounters::COUNT; ++iCounter)
261  {
262  uint64_t count = countersIter->second[iCounter];
263  outFile << m_counters.m_counterStringByCounter[iCounter] << std::setw(COUNT_WIDTH) << count << "\n";
264  totals[iCounter] += count;
265  }
266  }
267  }
268 
269  outFile << "=============\n";
270  outFile << "TOTALS\n";
271  outFile << "=============\n";
272  for (size_t iCounter = 0; iCounter < TCounters::COUNT; ++iCounter)
273  {
274  uint64_t count = totals[iCounter];
275  outFile << m_counters.m_counterStringByCounter[iCounter] << std::setw(COUNT_WIDTH) << count << "\n";
276  }
277 
278  outFile.close();
279  }

287  ThreadInfo* getThreadInfo()
288  {
289  if (s_tlThreadInfo.ptr == nullptr)
290  {
291  s_tlThreadInfo.ptr = new ThreadInfo;
292  }
293  if(s_tlThreadInfo.ptr->tid == 0)
294  {
295  s_tlThreadInfo.ptr->tid = ThisThread::getId();
296  Lock<mutex_type> lock(m_threadInfoMutex);
297  m_threadInfo.push_back(s_tlThreadInfo.ptr);
298  }
299  return s_tlThreadInfo.ptr;
300  }
