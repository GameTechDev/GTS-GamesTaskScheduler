234  void offerRange(typename TPartition::SplitterType const& splitter, TaskContext const& ctx)
235  {
236  // Split the range.
237  TRange::split_result splits;
238  m_range.split(splits, splitter);
239 
240  // Allocate the continuation.
241  Task* pContinuation = ctx.pMicroScheduler->allocateTask<TheftObserverTask>();
242  pContinuation->addRef(splits.size + 1, gts::memory_order::relaxed);
243  setContinuationTask(pContinuation);
244 
245  // Add the splits as siblings.
246  for (int iSibling = 0; iSibling < splits.size; ++iSibling)
247  {
248  Task* pSibling = ctx.pMicroScheduler->allocateTask<ParallelForTask>(
249  *this, splits.ranges[iSibling]);
250 
251  pContinuation->addChildTaskWithoutRef(pSibling);
252  ctx.pMicroScheduler->spawnTask(pSibling);
253  }
254 
255  // This task becomes a sibling.
256  pContinuation->addChildTaskWithoutRef(this);
257  }

260  void offerRange(TaskContext const& ctx, TRange& range, uint16_t depth = 0)
261  {
262  GTS_ASSERT(!range.empty() && "Bug in partitioner!");
263 
264  // Allocate the continuation.
265  Task* pContinuation = ctx.pMicroScheduler->allocateTask<TheftObserverTask>();
266  pContinuation->addRef(2, gts::memory_order::relaxed);
267  setContinuationTask(pContinuation);
268 
269  // The split is the right sibling.
270  Task* pSibling = ctx.pMicroScheduler->allocateTask<ParallelForTask>(*this, range, depth);
271  pContinuation->addChildTaskWithoutRef(pSibling);
272  ctx.pMicroScheduler->spawnTask(pSibling);
273 
274  // This task becomes a sibling.
