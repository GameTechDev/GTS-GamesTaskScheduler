61 void instrumentationGameLoop()
62 {
63  // To enable the macros, define GTS_ENABLE_CONCRT_LOGGER. In this example
64  // you can set the configuration to DebugWithInstrument or RelWithInstrument
65  // to create a log.
66 
67  // Init boilerplate
68  WorkerPool workerPool;
69  bool result = workerPool.initialize(2);
70  GTS_ASSERT(result);
71  MicroScheduler taskScheduler;
72  result = taskScheduler.initialize(&workerPool);
73  GTS_ASSERT(result);
74  ParallelFor parallelFor(taskScheduler);
75 
76  // In a game, you can dump per frame like so:
77 
78  char filename[32];
79  int frameId = 0;
80  while (true) // update loop.
81  {
82  // Mark the start of your frame.
83  GTS_TRACE_FRAME_MARK(analysis::CaptureMask::ALL);
84 
85  // Pretend workload so we can see some tasks.
86  uint32_t const elementCount = 1 << 10;
87  std::vector<char> vec(elementCount, 0);
88 
89  parallelFor(
90  BlockedRange1d<std::vector<char>::iterator>(vec.begin(), vec.end(), 1),
91  [](BlockedRange1d<std::vector<char>::iterator>& range, void*, TaskContext const&)
92  {
93 
94  },
95  AdaptivePartitioner(),
96  nullptr,
97  nullptr
98  );
99 
100  // Dump the frame data, where applicable.
101  sprintf_s(filename, "frame_%d.txt", frameId);
102  GTS_TRACE_DUMP(filename);
103 
104  ++frameId;
105 
106  break; // just an example, don't really loop.
107  }
108 
109 
110  // See: ConcurrentLogger.h for finer control over the logger.
111 
112  taskScheduler.shutdown();
113  workerPool.shutdown();
114 }
