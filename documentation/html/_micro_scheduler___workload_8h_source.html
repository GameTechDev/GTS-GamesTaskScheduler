113 class MicroSchedulerLambda_Workload : public MicroScheduler_Workload

115  template<typename TFunc, typename... TArgs>
116  static GTS_INLINE void _invoker(void* pRawTuple, WorkloadContext const& ctx)
117  {
118  // decode the tuple
119  using TupleType = std::tuple<TFunc, TArgs... >;
120  TupleType& tuple = *(TupleType*)pRawTuple;
121 
122  _invoke(tuple, std::make_integer_sequence<size_t, std::tuple_size<TupleType>::value>(), ctx);
123  }
124 
125  template<typename TTuple, size_t... Idxs>
126  static GTS_INLINE void _invoke(TTuple& tuple, std::integer_sequence<size_t, Idxs...>, WorkloadContext const& ctx)
127  {
128  // invoke function object packed in tuple
129  invoke<void>(std::move(std::get<Idxs>(tuple))..., ctx);
130  }
131 
132  using invoke_fn = void(*)(void*, WorkloadContext const&);
133 
134 public: // STRUCTORS:
135 
136  template<typename TFunc, typename... TArgs>
137  GTS_INLINE MicroSchedulerLambda_Workload(TFunc&& func, TArgs&&... args)
138  : m_invoker(&_invoker<TFunc, TArgs...>)
139  {
140  // Place the tuple at the end of this object.
141  new (this + 1) std::tuple<TFunc, TArgs... >(
142  std::forward<TFunc>(func), std::forward<TArgs>(args)...
143  );
144  }
145 
146 public: // MUTATORS:
147 
148  GTS_INLINE virtual void execute(WorkloadContext const& ctx) final
149  {
150  m_invoker(getTuple(), ctx);
151  }
152 
153 private:
154 
155  GTS_INLINE void* getTuple()
156  {
157  return this + 1;
158  }
