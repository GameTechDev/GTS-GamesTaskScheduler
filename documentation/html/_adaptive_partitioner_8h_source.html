27 class AdaptivePartitioner
28 {
31  // A deque of retained subdivided ranges that are ready to be offered to
32  // the scheduler when there is demand from other Workers. Ranges are executed
33  // off the back and shared off the front.
34  template<typename TRange, typename TSplitter>
35  class RetentionDeque
36  {
37  public:
38 
39  using split_result = typename TRange::split_result;
40 
41  //----------------------------------------------------------------------
42  GTS_INLINE RetentionDeque()
43  {}
44 
45  //----------------------------------------------------------------------
46  GTS_INLINE RetentionDeque(TRange const& range)
47  {
48  m_rangeRingBuf[m_back & MASK] = range;
49  m_depthRingBuf[m_back & MASK] = 0;
50  ++m_back;
51  }
52 
53  //----------------------------------------------------------------------
54  GTS_INLINE void popFront()
55  {
56  m_rangeRingBuf[m_front & MASK].~TRange();
57  ++m_front;
58  }
59 
60  //----------------------------------------------------------------------
61  GTS_INLINE void popBack()
62  {
63  m_rangeRingBuf[(m_back - 1) & MASK].~TRange();
64  --m_back;
65  }
66 
67  //----------------------------------------------------------------------
68  GTS_INLINE void pushBack(TRange& range, uint16_t depth)
69  {
70  m_rangeRingBuf[m_back & MASK] = range;
71  m_depthRingBuf[m_back & MASK] = depth;
72  ++m_back;
73  }
74 
75  //----------------------------------------------------------------------
76  GTS_INLINE TRange& front()
77  {
78  return m_rangeRingBuf[m_front & MASK];
79  }
80 
81  //----------------------------------------------------------------------
82  GTS_INLINE TRange& back()
83  {
84  return m_rangeRingBuf[(m_back - 1) & MASK];
85  }
86 
87  //----------------------------------------------------------------------
88  GTS_INLINE uint16_t frontDepth()
89  {
90  return m_depthRingBuf[m_front & MASK];
91  }
92 
93  //----------------------------------------------------------------------
94  GTS_INLINE uint16_t backDepth()
95  {
96  return m_depthRingBuf[(m_back - 1) & MASK];
97  }
98 
99  //----------------------------------------------------------------------
100  GTS_INLINE int16_t size() const
101  {
102  return m_back - m_front;
103  }
104 
105  //----------------------------------------------------------------------
106  GTS_INLINE bool empty() const
107  {
108  return size() == 0;
109  }
110 
111  //----------------------------------------------------------------------
112  GTS_INLINE bool full() const
113  {
114  return size() == MASK + 1;
115  }
116 
117  //----------------------------------------------------------------------
118  GTS_INLINE bool isDivisible(int16_t maxDepth)
119  {
120  return size() < MASK && backDepth() < maxDepth && back().isDivisible();
121  }
122 
123  //----------------------------------------------------------------------
124  GTS_NO_INLINE void splitToDepth(int16_t maxDepth)
125  {
126  TRange range = back();
127  uint16_t depth = backDepth();
128  popBack();
129 
130  while (size() - 1 + TRange::MAX_SPLITS < MASK
131  && depth < maxDepth
132  && range.isDivisible())
133  {
134  split_result splits;
135  range.split(splits, TSplitter());
136  ++depth;
137 
138  for (int ii = 0; ii < splits.size; ++ii)
139  {
140  m_rangeRingBuf[m_back & MASK] = splits.ranges[ii];
141  m_depthRingBuf[m_back & MASK] = depth;
142  ++m_back;
143  }
144  }
145  m_rangeRingBuf[m_back & MASK] = range;
146  m_depthRingBuf[m_back & MASK] = depth;
147  ++m_back;
148  }
149 
150  private:
151 
152  enum
153  {
154  MASK = gtsMax(TRange::SPLIT_FACTOR * 2, (size_t)8) - 1
155  };
156 
157  TRange m_rangeRingBuf[MASK + 1];
158  uint16_t m_depthRingBuf[MASK + 1];
159  int16_t m_front = 0;
160  int16_t m_back = 0;
161  };
162 
163 public:
164 
165  using splitter_type = EvenSplitter;
166 
167  //--------------------------------------------------------------------------
178  explicit GTS_INLINE AdaptivePartitioner(uint16_t initialMaxSplitDepth = 4)
179  : m_initialSplitDepth(0)
180  , m_maxBalancedSplitDepth(initialMaxSplitDepth)
181  , m_numSiblingThefts(0)
182  {}
183 
184  //--------------------------------------------------------------------------
185  template<typename TRange>
186  GTS_INLINE AdaptivePartitioner(AdaptivePartitioner& other, uint16_t depth, TRange const&)
187  : m_initialSplitDepth(getSplit<TRange>(other))
188  , m_maxBalancedSplitDepth(other.m_maxBalancedSplitDepth)
189  , m_numSiblingThefts(0)
190  {
191  GTS_ASSERT(depth <= m_maxBalancedSplitDepth);
192 
193  // Don't split past the depth of our sibling.
194  m_maxBalancedSplitDepth -= depth;
195  }

197  //--------------------------------------------------------------------------
198  template<typename TPattern, typename TRange>
199  GTS_INLINE Task* execute(TaskContext const& ctx, TPattern* pPattern, TRange& range)
200  {
201  // Divide up the range while its minimum size has not been reached. Spawn
202  // each division as new continuation and sibling tasks.
203  while (range.isDivisible() && isDivisible())
204  {
205  pPattern->initialOffer(ctx, range, splitter_type());
206  }
207 
208  return doExecute(ctx, pPattern, range, splitter_type());
209  }
210 
211  //--------------------------------------------------------------------------
212  template<typename TPattern, typename TRange>
213  GTS_INLINE void initialOffer(TaskContext const& ctx, TPattern* pPattern, TRange& range, splitter_type const& splitter)
214  {
215  pPattern->offerRange(ctx, range, splitter);
216  }
217 
218  //--------------------------------------------------------------------------
219  template<typename TPattern, typename TRange>
220  GTS_INLINE Task* doExecute(TaskContext const& ctx, TPattern* pPattern, TRange& range, splitter_type const& splitter)
221  {
222  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::RoyalBlue1, "ADAPTIVE doExecute", m_maxBalancedSplitDepth);
223 
224  if (!range.isDivisible() || m_maxBalancedSplitDepth == 0)
225  {
226  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::RoyalBlue1, "ADAPTIVE RUN", pPattern);
227 
228  // Execute any remaining work.
229  pPattern->run(ctx, range, splitter);
230  }
231  else
232  {
233  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::RoyalBlue2, "ADAPTIVE DEMAND", pPattern);
234  pPattern->balanceAndExecute(ctx, *this, range, splitter);
235  }
236 
237  GTS_SIM_TRACE_MARKER(sim_trace::MARKER_PARTITIONER_END);
238  return nullptr;
239  }
240 
241  //--------------------------------------------------------------------------
242  template<typename TPattern, typename TRange>
243  GTS_INLINE void balanceAndExecute(
244  TaskContext const& ctx,
245  TPattern* pPattern,
246  TRange& range,
247  splitter_type const& splitter)
248  {
249  RetentionDeque<TRange, splitter_type> retentionDeque(range);
250 
251  do
252  {
253  retentionDeque.splitToDepth(m_maxBalancedSplitDepth);
254 
255  // If the MicroScheduler has demand for work, try to offer work.
256  if (hasDemand<TRange>(ctx))
257  {
258  // Only offer a new task if there is more than one range in
259  // the deque.
260  if (retentionDeque.size() > 1)
261  {
262  pPattern->offerRange(ctx, retentionDeque.front(), retentionDeque.frontDepth());
263  retentionDeque.popFront();
264  continue;
265  }
266 
267  if (retentionDeque.isDivisible(m_maxBalancedSplitDepth))
268  {
269  continue;
270  }
271  }
272 
273  pPattern->run(ctx, retentionDeque.back(), splitter);
274  retentionDeque.popBack();
275 
276  } while (!retentionDeque.empty());
277  }
278 
279  //--------------------------------------------------------------------------
280  template<typename TRange>
281  GTS_INLINE bool hasDemand(TaskContext const& ctx)
282  {
283  GTS_TRACE_ZONE_MARKER_P0(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::PaleVioletRed3, "DOES THE SYSTEM HAVE DEMAND?");
284 
285  // Check for demand on the scheduler.
286  if (ctx.pMicroScheduler->hasDemand(true))
287  {
288  GTS_TRACE_ZONE_MARKER_P0(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::Green, "YES");
289  ++m_maxBalancedSplitDepth;
290  return true;
291  }
292 
293  return false;
294  }

296  //--------------------------------------------------------------------------
297  template<typename TRange>
298  GTS_INLINE void adjustIfStolen(Task* pThisTask)
299  {
300  // Only consider Tasks that are not part of the initial split phase.
301  if (m_initialSplitDepth == 0)
302  {
303  // Allow the Task to split to m_maxBalancedSplitDepth.
304  m_initialSplitDepth = 1;
305 
306  if (pThisTask->isStolen() &&
307  // Communicate the theft only if there is another child.
308  pThisTask->parent()->refCount() > 2)
309  {
310  GTS_TRACE_SCOPED_ZONE_P0(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::PaleVioletRed2, "NOFIFY STOLEN");
311 
312  if (m_maxBalancedSplitDepth == 0)
313  {
314  increaseDepth<TRange>();
315  };
316 
317  increaseDepth<TRange>();
318  }
319  }
320  }

325  {
326  GTS_ASSERT(workerCount > 0);
327  m_initialSplitDepth = TRange::adjustDivisor(nextPow2(workerCount), false);

344  //--------------------------------------------------------------------------
345  template<typename TRange>
346  GTS_INLINE void increaseDepth()
347  {
348  // Increment the depth if the number of sibling thefts is beyond the
349  // maximum split threshold. We do this so that a depth increment represents
350  // all children for the next depth.
351  if (++m_numSiblingThefts >= TRange::MAX_SPLITS)
352  {
353  ++m_maxBalancedSplitDepth;
354  m_numSiblingThefts = 0;
355  }
356  }
357 
358  //--------------------------------------------------------------------------
359  GTS_INLINE bool isDivisible()
360  {
361  return m_initialSplitDepth > 1;
362  }
363 
364  //--------------------------------------------------------------------------
365  GTS_INLINE uint16_t maxBalancedSplitDepth() const
366  {
367  return m_maxBalancedSplitDepth;
368  }
