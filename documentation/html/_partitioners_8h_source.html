209 class AdaptivePartitioner
210 {
211 public:
212 
213  using splitter_type = EvenSplitter;
214 
215  //--------------------------------------------------------------------------
226  explicit GTS_INLINE AdaptivePartitioner(uint16_t initialMaxSplitDepth = 5)
227  : m_initialSplitDepth(0)
228  , m_maxBalancedSplitDepth(initialMaxSplitDepth)
229  , m_numSiblingThefts(0)
230  {}
231 
232  //--------------------------------------------------------------------------
233  template<typename TRange>
234  GTS_INLINE AdaptivePartitioner(AdaptivePartitioner& partitioner, uint16_t depth, TRange const&)
235  : m_initialSplitDepth(getSplit<TRange>(partitioner))
236  , m_maxBalancedSplitDepth(partitioner.m_maxBalancedSplitDepth)
237  , m_numSiblingThefts(0)
238  {
239  GTS_ASSERT(depth <= m_maxBalancedSplitDepth);
240 
241  // Don't split past the depth of our sibling. This keep the task tree
242  // balanced.
243  m_maxBalancedSplitDepth -= depth;
244  }
245 
246  //--------------------------------------------------------------------------
247  template<typename TPattern, typename TRange>
248  GTS_INLINE Task* execute(TaskContext const& ctx, TPattern* pPattern, TRange& range)
249  {
250  // Divide up the range while its minimum size has not been reached. Queue
251  // each division as new continuation and sibling tasks.
252  while (range.isDivisible() && isDivisible())
253  {
254  pPattern->initialOffer(ctx, range, splitter_type());
255  }
256 
257  return doExecute(ctx, pPattern, range, splitter_type());
258  }
259 
260  //--------------------------------------------------------------------------
261  template<typename TPattern, typename TRange>
262  GTS_INLINE void initialOffer(TaskContext const& ctx, TPattern* pPattern, TRange& range, splitter_type const& splitter)
263  {
264  pPattern->offerRange(ctx, range, splitter);
265  }
266 
267  //--------------------------------------------------------------------------
268  template<typename TPattern, typename TRange>
269  GTS_INLINE Task* doExecute(TaskContext const& ctx, TPattern* pPattern, TRange& range, splitter_type const& splitter)
270  {
271  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::RoyalBlue1, "ADAPTIVE doExecute", m_maxBalancedSplitDepth);
272 
273  // Split the range if there is a remaining initial split.
274  if (m_initialSplitDepth == 1)
275  {
276  pPattern->balanceAndExecute(ctx, *this, range, splitter);
277  }
278  else
279  {
280  // Execute any remaining work.
281  pPattern->run(ctx, range, splitter);
282  }
283 
284  return nullptr;
285  }
286 
287  //--------------------------------------------------------------------------
288  template<typename TObserverTask, typename TPattern, typename TRange>
289  GTS_INLINE void balanceAndExecute(
290  TaskContext const& ctx,
291  TPattern* pPattern,
292  TRange& range,
293  splitter_type const& splitter)
294  {
295  GTS_ASSERT(m_initialSplitDepth == 1);
296 
297  m_initialSplitDepth = 0;
298  uint16_t splitDepth = 0;
299 
300  // Split to depth to balance the Task tree.
301  while (splitDepth < m_maxBalancedSplitDepth && range.isDivisible())
302  {
303  GTS_TRACE_SCOPED_ZONE_P0(analysis::CaptureMask::MICRO_SCHEDULER_PROFILE, analysis::Color::PaleVioletRed4, "ADAPTIVE SPLIT");
304 
305  typename TRange::split_result splits;
306  range.split(splits, splitter);
307  ++splitDepth;
308 
309  for (int ii = 0; ii < splits.size; ++ii)
310  {
311  pPattern->offerRange(ctx, splits.ranges[ii], splitDepth);
312  }
313  }
314 
315  // Execute any remaining work.
316  pPattern->run(ctx, range, splitter);
317  }

327  //--------------------------------------------------------------------------
328  template<typename TObserverTask, typename TRange>
329  GTS_INLINE void adjustIfStolen(Task* pThisTask)
330  {
331  // Only consider Tasks that are not part of the initial split phase.
332  if (m_initialSplitDepth == 0)
333  {
334  // Allow the Task to split to m_maxBalancedSplitDepth.
335  m_initialSplitDepth = 1;
336 
337  if (pThisTask->isStolen() &&
338  // Communicate the theft only if there is another child.
339  pThisTask->parent()->refCount() > 2)
340  {
341  TObserverTask* pObserver = (TObserverTask*)pThisTask->parent();
342  pObserver->m_childTaskStolen.exchange(true, memory_order::acq_rel);
343 
344  ++m_numSiblingThefts;
345  if (m_maxBalancedSplitDepth == 0)
346  {
347  ++m_numSiblingThefts;
348  };
349 
350  adjustDepth<TRange>();
351  }
352  }
353  }

376  //--------------------------------------------------------------------------
377  template<typename TRange>
378  GTS_INLINE void adjustDepth()
379  {
380  // Increment the depth if the number of sibling thefts is beyond the
381  // maximum split threshold. We do this so that a depth increment represents
382  // all children for the next depth.
383  if (m_numSiblingThefts >= TRange::MAX_SPLITS)
384  {
385  m_maxBalancedSplitDepth++;
386  m_numSiblingThefts = 0;
387  }
388  }
389 
390  //--------------------------------------------------------------------------
391  GTS_INLINE bool isDivisible()
392  {
393  return m_initialSplitDepth > 1;
394  }

397  GTS_INLINE uint16_t maxBalancedSplitDepth() const
398  {
399  return m_maxBalancedSplitDepth;
400  }
