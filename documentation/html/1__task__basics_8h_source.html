54 void taskBasics()
55 {
56  printf ("================\n");
57  printf ("taskBasics\n");
58  printf ("================\n");
59 
60  // Init boilerplate
61  WorkerPool workerPool;
62  bool result = workerPool.initialize();
63  GTS_ASSERT(result);
64  MicroScheduler microScheduler;
65  result = microScheduler.initialize(&workerPool);
66  GTS_ASSERT(result);
67 
68  // Create a new task.
69  int val = 1;
70  Task* pTask = microScheduler.allocateTask<BasicTask>(val);
71 
72  // Tasks start with a refCount of 1.
73  GTS_ASSERT(pTask->refCount() == 1);
74 
75  // Let's say we only want to run this task on the master worker thread.
76  pTask->setAffinity(0);
77  // Now this task can only every be run on worker thread #0.
78 
79  // Spawn the task for execution and wait for it to complete. The act of
80  // spawning a task implicitly reduces its refCount by 1 after execution.
81  // If the refCount it zero after execution, the task will be destroyed,
82  // which is the desired behavior (in general).
83  microScheduler.spawnTaskAndWait(pTask /*, optional priority*/);
84  // NOTE: wait ^^^^ does NOT mean that this thread is blocked. This thread will
85  // actively execute any tasks in the scheduler until pTask completes.
86 
87  microScheduler.shutdown();
88  workerPool.shutdown();
89 }
