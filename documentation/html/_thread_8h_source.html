104  class AffinitySet
105  {
106  #if GTS_64BIT
107  static constexpr uint32_t MAX_SETS = 16;
108  #else
109  static constexpr uint32_t MAX_SETS = 32;
110  #endif
111 
112  public:
113  GTS_INLINE void set(uint32_t cpuId)
114  {
115  GTS_ASSERT(cpuId < sizeof(unsigned long) * MAX_SETS * 8);
116  m_bitset[cpuId / MAX_SETS] |= 1 << (cpuId % MAX_SETS);
117  }
118  GTS_INLINE void combine(AffinitySet const& other)
119  {
120  for(uint32_t ii = 0; ii < MAX_SETS; ++ii)
121  {
122  m_bitset[ii] |= other.m_bitset[ii];
123  }
124  }
125  GTS_INLINE bool empty() const
126  {
127  for(uint32_t ii = 0; ii < MAX_SETS; ++ii)
128  {
129  if(m_bitset[ii] != 0)
130  {
131  return false;
132  }
133  }
134  return true;
135  }
136  private:
137  GTS_ALIGN(sizeof(void*)) size_t m_bitset[MAX_SETS] = {0};
138  };
