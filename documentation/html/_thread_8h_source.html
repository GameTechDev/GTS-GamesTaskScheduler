589 enum class BackoffGrowth
590 {
592  Arithmetic,
594  Geometric
595 };

597 //------------------------------------------------------------------------------
602 GTS_INLINE void pauseFor(uint32_t cycleCount)
603 {
604  int32_t count = 0;
605 
606  uint64_t prev = GTS_RDTSC();
607  const uint64_t finish = prev + cycleCount;
608  do
609  {
610  spin(count);
611  count *= 2;
612 
613  uint64_t curr = GTS_RDTSC();
614  if (curr <= prev)
615  {
616  // Quit on context switch or overflow.
617  break;
618  }
619  prev = curr;
620 
621  } while (prev < finish);
622 }

635 template<BackoffGrowth GROWTH_RATE = BackoffGrowth::Geometric, bool YIELD = true>
636 class Backoff
637 {
638 public:
639 
640  //--------------------------------------------------------------------------
645  GTS_INLINE bool tick()
646  {
647  if(!YIELD || m_cyclesSoFar < GTS_CONTEXT_SWITCH_CYCLES)
648  {
649  uint64_t begin = GTS_RDTSC();
650  spin(m_count);
651  uint64_t end = GTS_RDTSC();
652 
653  if (end <= begin)
654  {
655  // Context switch or overflow occurred. Assume we are ready to yield.
656  m_cyclesSoFar = GTS_CONTEXT_SWITCH_CYCLES;
657  }
658  else
659  {
660  m_cyclesSoFar += end - begin;
661  }
662 
663  switch (GROWTH_RATE)
664  {
665  case BackoffGrowth::Arithmetic:
666  m_count += 1;
667  break;
668  case BackoffGrowth::Geometric:
669  m_count *= 2;
670  break;
671  }
672  return false;
673  }
674  else if(YIELD)
675  {
676  GTS_TRACE_SCOPED_ZONE_P0(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkMagenta, "YIELD");
677  ThisThread::yield();
678  }
679 
680  return true;
681  }
682 
683  //--------------------------------------------------------------------------
688  GTS_INLINE void reset()
689  {
690  m_count = 1;
691  m_cyclesSoFar = 0;
692  }
693 
694  //--------------------------------------------------------------------------
701  GTS_INLINE int32_t spinCount() const
702  {
703  return m_count;
704  }
705 
706 private:
707 
708  uint64_t m_cyclesSoFar = 0;
709  int32_t m_count = 1;
710 };

754 template<typename TBackoff = Backoff<BackoffGrowth::Geometric, true>>
755 class UnfairSpinMutex
756 {
757  // Backoff spinlock
758 public:
759 
760  using backoff_type = TBackoff;
761 
762 public:
763 
764  //--------------------------------------------------------------------------
765  GTS_INLINE UnfairSpinMutex() : m_lock(AVAILABLE) {}
766 
767  //--------------------------------------------------------------------------
768  GTS_INLINE UnfairSpinMutex(UnfairSpinMutex&& other)
769  : m_lock(std::move(other.m_lock.load(memory_order::acquire)))
770  {}
771 
772  //--------------------------------------------------------------------------
773  GTS_INLINE UnfairSpinMutex& operator=(UnfairSpinMutex&& other)
774  {
775  if (this != &other)
776  {
777  m_lock.load(std::move(other.m_lock.load(memory_order::acquire)), memory_order::relaxed);
778  }
779  return *this;
780  }
781 
782  //--------------------------------------------------------------------------
783  GTS_INLINE bool try_lock()
784  {
785  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkRed, "TRY LOCK", this);
786 
787  // Test and test-and-set.
788  return m_lock.load(memory_order::relaxed) == AVAILABLE &&
789  // Only perform the expensive exchange if the lock seems available.
790  m_lock.exchange(LOCKED, memory_order::acq_rel) == AVAILABLE;
791  }
792 
793  //--------------------------------------------------------------------------
794  GTS_INLINE void lock()
795  {
796  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkRed, "LOCKING", this);
797 
798  backoff_type backoff;
799  while (!try_lock())
800  {
801  backoff.tick();
802  }
803  }
804 
805  //--------------------------------------------------------------------------
806  GTS_INLINE void unlock()
807  {
808  m_lock.store(AVAILABLE, memory_order::release);
809  }
810 
811  //--------------------------------------------------------------------------
812  GTS_INLINE bool isLocked() const
813  {
814  return m_lock.load(memory_order::acquire) == LOCKED;
815  }
816 
817 private:
818 
819  enum { AVAILABLE = 0, LOCKED = 1 };
820 
821  Atomic<uint8_t> m_lock;
822 };

831 template<typename TBackoff = Backoff<BackoffGrowth::Geometric, true>>
832 class FairSpinMutex
833 {
834  // Ticket spinlock
835 public:
836 
837  using backoff_type = TBackoff;
838 
839 public:
840 
841  //--------------------------------------------------------------------------
842  GTS_INLINE FairSpinMutex() : m_nextTicket(0), m_currTicket(0) {}
843 
844  //--------------------------------------------------------------------------
845  GTS_INLINE FairSpinMutex(FairSpinMutex&& other)
846  : m_nextTicket(std::move(other.m_nextTicket.load(memory_order::acquire)))
847  , m_currTicket(std::move(other.m_currTicket.load(memory_order::acquire)))
848  {}
849 
850  //--------------------------------------------------------------------------
851  GTS_INLINE FairSpinMutex& operator=(FairSpinMutex&& other)
852  {
853  if (this != &other)
854  {
855  m_nextTicket.load(std::move(other.m_nextTicket.load(memory_order::acquire)), memory_order::relaxed);
856  m_currTicket.load(std::move(other.m_currTicket.load(memory_order::acquire)), memory_order::relaxed);
857  }
858  return *this;
859  }
860 
861  //--------------------------------------------------------------------------
862  GTS_INLINE bool try_lock()
863  {
864  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkRed, "TRY LOCK", this);
865 
866  uint16_t currTicket = m_currTicket.load(memory_order::acquire);
867  uint16_t myTicket = m_nextTicket.load(memory_order::acquire);
868 
869  if (currTicket == myTicket)
870  {
871  return m_nextTicket.compare_exchange_weak(myTicket, myTicket + 1, memory_order::acq_rel);
872  }
873 
874  return false;
875  }
876 
877  //--------------------------------------------------------------------------
878  GTS_INLINE void lock()
879  {
880  GTS_TRACE_SCOPED_ZONE_P1(analysis::CaptureMask::THREAD_PROFILE, analysis::Color::DarkRed, "LOCKING", this);
881 
882  // Get the next ticket.
883  uint16_t myTicket = m_nextTicket.fetch_add(1, memory_order::acquire);
884 
885  // Wait until my ticket is up.
886  backoff_type backoff;
887  while (m_currTicket.load(memory_order::acquire) != myTicket)
888  {
889  backoff.tick();
890  }
891  }
892 
893  //--------------------------------------------------------------------------
894  GTS_INLINE void unlock()
895  {
896  // Mark the next ticket as ready.
897  m_currTicket.fetch_add(1, memory_order::release);
898  }
899 
900  //--------------------------------------------------------------------------
901  GTS_INLINE bool isLocked() const
902  {
903  return m_currTicket.load(memory_order::acquire) == m_nextTicket.load(memory_order::acquire);
904  }
905

951

953  {

1049  }
