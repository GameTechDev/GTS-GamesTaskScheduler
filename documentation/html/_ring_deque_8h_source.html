396  bool _grow(size_type size)
397  {
398  if (size < MIN_CAPACITY || size > MAX_CAPACITY)
399  {
400  // can't grow anymore.
401  return false;
402  }
403 
404  value_type* pOldBuffer = m_pRingBuffer;
405  size_type oldMask = (m_capacity - 1);
406 
407  m_pRingBuffer = allocator_type::template allocate<value_type>((size_t)size);
408  m_capacity = size;
409 
410  // Copy the elements into the new buffer.
411  for (size_type ii = m_front; ii != m_back; ++ii)
412  {
413  allocator_type::construct(m_pRingBuffer + (ii & (m_capacity - 1)), pOldBuffer[ii & oldMask]);
414  }
415 
416  allocator_type::deallocate(pOldBuffer, (size_t)oldMask + 1);
417 
418  return true;
419  }

421  void _deepCopy(RingDeque const& src)
422  {
423  RingDeque* pSrcRingBuffer = src.m_pRingBuffer;
424  if (pSrcRingBuffer)
425  {
426  // Create a RingDeque
427  m_pRingBuffer = allocator_type::template allocate<value_type>(pSrcRingBuffer->m_capacity);
428  m_capacity = pSrcRingBuffer->m_capacity;
429 
430  //
431  // Copy RingDeque elements.
432 
433  size_type f = src.m_front.load(memory_order::relaxed);
434  size_type b = src.m_back.load(memory_order::relaxed);
435 
436  for (size_type ii = f; ii != b; ++ii)
437  {
438  allocator_type::construct(pRingBuffer + (ii & (m_capacity - 1)), pSrcRingBuffer[ii & (pSrcRingBuffer->m_capacity - 1)]);
439  }
440

443  }
