33 void heteroInit()
34 {
35  // This example demonstrates hierarchical Task scheduling over two Efficiency Classes. Each Efficiency
36  // class gets it's own Worker Pool and Micro-scheduler. The goal of this configuration is to keep Tasks
37  // local to their Efficiency Class as long as possible. The only non-local communication that happens
38  // is when the more efficient Micro-scheduler victimizes the less efficient Micro-scheduler. Victimization
39  // is one-way, because we don't want a more computationally intensive Task to run on a less efficient processor,
40  // which could dramatically increase the overall makespan of the schedule.
41 
42  // Get the CPU topology.
43  SystemTopology sysTopo;
44  Thread::getSystemTopology(sysTopo);
45 
46  // Get the max efficiency.
47  uint32_t maxEfficiency = 0;
48  for (size_t iGroup = 0; iGroup < sysTopo.groupInfoElementCount; ++iGroup)
49  {
50  ProcessorGroupInfo const& groupInfo = sysTopo.pGroupInfoArray[iGroup];
51 
52  // Loop through each physical core.
53  for (size_t iCore = 0; iCore < groupInfo.coreInfoElementCount; ++iCore)
54  {
55  maxEfficiency = gtsMax(maxEfficiency, (uint32_t)groupInfo.pCoreInfoArray[iCore].efficiencyClass);
56  }
57  }
58 
59  std::vector<std::vector<WorkerPool*>> workerPoolsByEfficiency(maxEfficiency + 1);
60  std::vector<std::vector<MicroScheduler*>> microSchedulersByEfficiency(maxEfficiency + 1);
61 
62  // Loop through each processor group.
63  for (size_t iGroup = 0; iGroup < sysTopo.groupInfoElementCount; ++iGroup)
64  {
65  ProcessorGroupInfo const& groupInfo = sysTopo.pGroupInfoArray[iGroup];
66 
67  std::vector<size_t> numThreadsByEfficiencyClass(maxEfficiency + 1);
68  std::vector<AffinityType> affinityByEfficiencyClass(maxEfficiency + 1);
69 
70  // Loop through each physical core.
71  for (size_t iCore = 0; iCore < groupInfo.coreInfoElementCount; ++iCore)
72  {
73  CpuCoreInfo const& coreInfo = groupInfo.pCoreInfoArray[iCore];
74 
75  // Loop through each thread.
76  for (size_t iThead = 0; iThead < coreInfo.hardwareThreadCount; ++iThead)
77  {
78  numThreadsByEfficiencyClass[coreInfo.efficiencyClass]++;
79  affinityByEfficiencyClass[coreInfo.efficiencyClass] |= coreInfo.pLogicalAffinityMasks[iThead];
80  }
81  }
82 
83  //
84  // Create the WorkerPool for each efficiency class.
85 
86  for (size_t ii = 0; ii <= maxEfficiency; ++ii)
87  {
88  WorkerPoolDesc workerPoolDesc;
89 
90  if (iGroup == 0 && ii == maxEfficiency)
91  {
92  // The Mater thread IS part of this class.
93 
94  // Affinitize the Master thread here since the Worker Pool does not own it.
95  gts::ThisThread::setAffinity(0, sysTopo.pGroupInfoArray[0].pCoreInfoArray[0].pLogicalAffinityMasks[0]);
96  }
97  else
98  {
99  // The Mater thread IS NOT part of this node.
100 
101  // Add filler desc for the master so that Worker threads
102  // are created for all the threads in the class.
103  workerPoolDesc.workerDescs.push_back(WorkerThreadDesc());
104  }
105 
106  size_t numThreads = numThreadsByEfficiencyClass[ii];
107  for (size_t iWorker = 0; iWorker < numThreads; ++iWorker)
108  {
109  WorkerThreadDesc workerDesc;
110  workerDesc.affinity = {iGroup, affinityByEfficiencyClass[ii]};
111  workerPoolDesc.workerDescs.push_back(workerDesc);
112  }
113 
114  WorkerPool* pWorkerPool = new WorkerPool;
115  bool result = pWorkerPool->initialize(workerPoolDesc);
116  GTS_ASSERT(result);
117  workerPoolsByEfficiency[ii].push_back(pWorkerPool);
118  }
119  }
120 
121  // Create the MicroSchedulers
122  for (size_t ii = 0; ii <= maxEfficiency; ++ii)
123  {
124  auto& vec = workerPoolsByEfficiency[ii];
125 
126  for (auto& wp : vec)
127  {
128  MicroScheduler* pMicroScheduler = new MicroScheduler;
129  pMicroScheduler->initialize(wp);
130  microSchedulersByEfficiency[ii].push_back(pMicroScheduler);
131  }
132  }
133 
134  // Link such that higher efficiency schedulers victimize lower efficiency schedulers.
135  for (int ii = maxEfficiency; ii > 0; --ii)
136  {
137  auto highVec = microSchedulersByEfficiency[ii];
138 
139  for (auto highSched : highVec)
140  {
141  for (int jj = ii; jj >= 0; --jj)
142  {
143  auto lowVec = microSchedulersByEfficiency[jj];
144 
145  for (auto lowSched : lowVec)
146  {
147  highSched->addExternalVictim(lowSched);
148  }
149  }
150  }
151  }
152 
153  // schedule stuff ...
154 
155 }
