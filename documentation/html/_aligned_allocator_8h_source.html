48 template <size_t ALIGNMENT>
49 class AlignedAllocator
50 {
51  static_assert(ALIGNMENT >= sizeof(uintptr_t), "ALIGNMENT to small");
52  static_assert(isPow2(ALIGNMENT), "ALIGNMENT must be a power of 2.");

61  //--------------------------------------------------------------------------
62  template<typename T>
63  size_t max_size() const
64  {
65  return SIZE_MAX / sizeof(T);
66  }
67 
68  //--------------------------------------------------------------------------
69  template<typename T, typename... TArgs>
70  void construct(T* ptr, TArgs&&... args)
71  {
72  void* const pv = static_cast<void*>(ptr);
73  new (pv) T(std::forward<TArgs>(args)...);
74  }
75 
76  //--------------------------------------------------------------------------
77  template<typename T>
78  void destroy(T* const ptr)
79  {
80  ptr->~T();
81  }
82 
83  //--------------------------------------------------------------------------
84  template<typename T>
85  T* allocate(const size_t n, size_t align = ALIGNMENT) const
86  {
87  if (n == 0)
88  {
89  return nullptr;
90  }
91 
92  if (n > max_size<T>())
93  {
94  GTS_ASSERT(0 && "Integer overflow.");
95  return nullptr;
96  }
97 
98  void * const pv = GTS_ALIGNED_MALLOC(n * sizeof(T), align);
99  if (pv == nullptr)
100  {
101  GTS_ASSERT(0 && "Failed Allocation.");
102  return nullptr;
103  }
104 
105  return static_cast<T *>(pv);
106  }
107 
108  //--------------------------------------------------------------------------
109  template<typename T>
110  void deallocate(T* const ptr, size_t) const
111  {
112  if(ptr)
113  {
114  GTS_ALIGNED_FREE(ptr);
115  }
116  }
117 
118  //--------------------------------------------------------------------------
119  template<typename T, typename... TArgs>
120  T* new_object(TArgs&&... args)
121  {
122  T* ptr = allocate<T>(1);
123  construct(ptr, std::forward<TArgs>(args)...);
124  return ptr;
125  }
126 
127  //--------------------------------------------------------------------------
128  template<typename T>
129  void delete_object(T* ptr)
130  {
131  destroy(ptr);
132  deallocate(ptr, sizeof(T));
133  }
134 
135  //--------------------------------------------------------------------------
136  template<typename T>
137  T* vector_new_object(size_t count)
138  {
139  T* ptr = allocate<T>(count);
140  for (size_t ii = 0; ii < count; ++ii)
141  {
142  construct(ptr + ii);
143  }
144  return ptr;
145  }
146 
147  //--------------------------------------------------------------------------
148  template<typename T>
149  void vector_delete_object(T* ptr, size_t count)
150  {
151  for (size_t ii = 0; ii < count; ++ii)
152  {
153  destroy(ptr + ii);
154  }
155  deallocate(ptr, sizeof(T) * count);
156  }
157 };
