48  // https://docs.microsoft.com/en-us/cpp/intrinsics/cpuid-cpuidex?view=msvc-160
49  class CpuId
50  {
51  public:
52  CpuId()
53  : m_f_1_ECX{ 0 }
54  , m_f_1_EDX{ 0 }
55  , m_f_7_EBX{ 0 }
56  , m_f_7_ECX{ 0 }
57  , m_f_81_ECX{ 0 }
58  , m_f_81_EDX{ 0 }
59  {
60 #if GTS_ARCH_X86
61 
62  Vector<Array<int, 4>> data;
63  Array<int, 4> cpuInfo;
64 
65  // Calling __cpuid with 0x0 as the function_id argument
66  // gets the number of the highest valid function ID.
67  __cpuid(cpuInfo.data, 0);
68  int numIds = cpuInfo.data[0];
69 
70  for (int i = 0; i <= numIds; ++i)
71  {
72  __cpuid(cpuInfo.data, i);
73  data.push_back(cpuInfo);
74  }
75 
76  // load bitset with flags for function 0x00000001
77  if (numIds >= 1)
78  {
79  m_f_1_ECX = data[1].data[2];
80  m_f_1_EDX = data[1].data[3];
81  }
82 
83  // load bitset with flags for function 0x00000007
84  if (numIds >= 7)
85  {
86  m_f_7_EBX = data[7].data[1];
87  m_f_7_ECX = data[7].data[2];
88  }
89 
90  data.clear();
91 
92  // Calling __cpuid with 0x80000000 as the function_id argument
93  // gets the number of the highest valid extended ID.
94  __cpuid(cpuInfo.data, 0x80000000);
95  int numExIds = cpuInfo.data[0];
96 
97  for (int i = 0x80000000; i <= numExIds; ++i)
98  {
99  __cpuidex(cpuInfo.data, i, 0);
100  data.push_back(cpuInfo);
101  }
102 
103  // load bitset with flags for function 0x80000001
104  if (numExIds >= 0x80000001)
105  {
106  m_f_81_ECX = data[1].data[2];
107  m_f_81_EDX = data[1].data[3];
108  }
109 #endif
110  };
111 
112  uint32_t m_f_1_ECX;
113  uint32_t m_f_1_EDX;
114  uint32_t m_f_7_EBX;
115  uint32_t m_f_7_ECX;
116  uint32_t m_f_81_ECX;
117  uint32_t m_f_81_EDX;
118  };
