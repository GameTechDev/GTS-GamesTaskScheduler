39 inline void initTelemetry()
40 {
41  // tmInitialize gets Telemetry started. You can pass a pointer to the memory
42  // that you want Telemetry to use, or call tmInitialize(0, 0) and Telemetry
43  // will allocate memory from the heap. Increasing memory gives telemetry
44  // more internal buffers and helps for captures that are heavily marked up.
45  tm_int32 telemetry_memory_size = 8 * 1024 * 1024;
46  char* telemetry_memory = new char[telemetry_memory_size];
47  tmInitialize(telemetry_memory_size, telemetry_memory);
48 
49  // Connect to a telemetry server and start profiling
50  tm_error err = tmOpen(
51  0, // unused
52  "GTS", // program name, don't use slashes or weird character that will screw up a filename
53  __DATE__ " " __TIME__, // identifier, could be date time, or a build number ... whatever you want
54  "localhost", // telemetry server address
55  TMCT_TCP, // network capture
56  4719, // telemetry server port
57  TMOF_INIT_NETWORKING, // flags
58  100); // timeout in milliseconds ... pass -1 for infinite
59 
60  if (err == TMERR_DISABLED)
61  {
62  GTS_ASSERT(0 && "Telemetry is disabled via #define NTELEMETRY");
63  }
64  else if (err == TMERR_UNINITIALIZED)
65  {
66  GTS_ASSERT(0 && "tmInitialize failed or was not called");
67  }
68  else if (err == TMERR_NETWORK_NOT_INITIALIZED)
69  {
70  GTS_ASSERT(0 && "WSAStartup was not called before tmOpen! Call WSAStartup or pass TMOF_INIT_NETWORKING.");
71  }
72  else if (err == TMERR_NULL_API)
73  {
74  GTS_ASSERT(0 && "There is no Telemetry API (the DLL isn't in the EXE's path)!");
75  }
76  else if (err == TMERR_COULD_NOT_CONNECT)
77  {
78  GTS_ASSERT(0 && "There is no Telemetry server running");
79  }
80 }
