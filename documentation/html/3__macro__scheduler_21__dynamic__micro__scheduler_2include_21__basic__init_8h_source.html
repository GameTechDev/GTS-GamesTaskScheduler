38 void initDynamicMicroScheduler()
39 {
40  printf("================\n");
41  printf("Init Dynamic Micro-scheduler\n");
42  printf("================\n");
43 
44  // A MacroScheduler is a high level scheduler that maps a persistent task
45  // graph (DAG) of Nodes to set of ComputeResources. A DynamicMicroScheduler
46  // is a MacroScheduler that executes a DAG exclusively on one or more
47  // MicroSchedulers. Each Node is converted to a Task when it is ready to be
48  // executed.
49 
50  // For this example, let's say we have two schedulers, one with higher
51  // priority than the other.
52 
53  //
54  // First, we create the MicroSchedulders and map them to
55  // MicroScheduler_ComputeResources that can be consumed by
56  // a CentralQueue_MacroScheduler.
57 
58  constexpr uint32_t PRIORITY_COUNT = 2;
59  WorkerPool workerPool[PRIORITY_COUNT];
60  MicroScheduler microScheduler[PRIORITY_COUNT];
61  MicroScheduler_ComputeResource microSchedulerCompResource[PRIORITY_COUNT];
62 
63  for (uint32_t iPriority = 0; iPriority < PRIORITY_COUNT; ++iPriority)
64  {
65  WorkerPoolDesc workerPoolDesc;
66 
67  for (uint32_t iWorker = 0; iWorker < Thread::getHardwareThreadCount(); iWorker++)
68  {
69  WorkerThreadDesc workerDesc;
70  workerDesc.priority = (Thread::Priority)((uint32_t)Thread::Priority::PRIORITY_NORMAL + iPriority);
71  workerPoolDesc.workerDescs.push_back(workerDesc);
72  }
73 
74  workerPool[iPriority].initialize(workerPoolDesc);
75  microScheduler[iPriority].initialize(&workerPool[iPriority]);
76 
77  microSchedulerCompResource[iPriority].init(microScheduler + iPriority);
78  }
79 
80  //
81  // Second, we create a MicroScheduler_ComputeResources and map the
82  // MicroScheduler_ComputeResources to it.
83 
84  MacroSchedulerDesc generalizedDagSchedulerDesc;
85  for (uint32_t iPriority = 0; iPriority < PRIORITY_COUNT; ++iPriority)
86  {
87  generalizedDagSchedulerDesc.computeResources.push_back(microSchedulerCompResource + iPriority);
88  }
89 
90  MacroScheduler* pMacroScheduler = new CentralQueue_MacroScheduler;
91  pMacroScheduler->init(generalizedDagSchedulerDesc);
92 
93  //
94  // For this example let's revisit the simple DAG from the Quick Start
95  // example. However, this time we want A, B, and D to run on the high priority
96  // scheduler and C to run on the low priority scheduler. We do this with
97  // Node affinity.
98  /*
99  +---+
100  +---->| B |-----+
101  | +---+ |
102  | v
103  +---+ +---+
104  | A | | D |
105  +---+ +---+
106  | ^
107  | +---+ |
108  +---->| C |-----+
109  +---+
110  */
111 
112 
113  Node* pA = pMacroScheduler->allocateNode();
114  pA->addWorkload<MicroSchedulerLambda_Workload>([](WorkloadContext const&) { printf("A\n"); });
115  pA->setAffinity(microSchedulerCompResource[1].id());
116 
117  Node* pB = pMacroScheduler->allocateNode();
118  pB->addWorkload<MicroSchedulerLambda_Workload>([](WorkloadContext const&) { printf("B\n"); });
119  pB->setAffinity(microSchedulerCompResource[1].id());
120 
121  Node* pC = pMacroScheduler->allocateNode();
122  pC->addWorkload<MicroSchedulerLambda_Workload>([](WorkloadContext const&) { printf("C\n"); });
123  pC->setAffinity(microSchedulerCompResource[0].id());
124 
125  Node* pD = pMacroScheduler->allocateNode();
126  pD->addWorkload<MicroSchedulerLambda_Workload>([](WorkloadContext const&) { printf("D\n"); });
127  pD->setAffinity(microSchedulerCompResource[1].id());
128 
129  pA->addSuccessor(pB);
130  pA->addSuccessor(pC);
131  pB->addSuccessor(pD);
132  pC->addSuccessor(pD);
133 
134  // Build and execute the schedule
135  Schedule* pSchedule = pMacroScheduler->buildSchedule(pA, pD);
136  pMacroScheduler->executeSchedule(pSchedule, microSchedulerCompResource[1].id());
137 
138  pMacroScheduler->freeSchedule(pSchedule);
139  delete pMacroScheduler;
140 
141  printf("SUCCESS!\n\n");
142 }
