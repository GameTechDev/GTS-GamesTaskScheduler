40 void initCatsForHybridCpu()
41 {
42  printf("================\n");
43  printf("Init CATS Macro-scheduler a Hybrid CPU \n");
44  printf("================\n");
45 
46  // A MacroScheduler is a high level scheduler that maps a persistent task
47  // graph (DAG) of Nodes to set of ComputeResources. A CriticallyAware_MacroScheduler
48  // is a MacroScheduler that executes a DAG on a heterogeneous set of
49  // ComputeResources. It uses a heuristic that forces the critical path of
50  // a DAG to execute on the most efficient ComputeResource, which minimizes
51  // the span of the execution.
52  //
53  // Nodes are made available to ComputeResources via Workload types associated
54  // with each ComputeResource type. For instance, if a Node can execute on
55  // ComputeResouces A, B, and C, then it must have Workloads attached to it that
56  // can run on A, B, and C.
57 
58  // NOTE: this example assumes a hybrid CPU two types of processors.
59 
60  //
61  // First, we create the MicroSchedulders and map them to
62  // DynamicMicroScheduler_ComputeResources that can be consumed by
63  // a DynamicMicroScheduler_MacroScheduler.
64  //
65  // Each MicroSchedulder cover a homogeneous set of processors on the CPU.
66 
67  // Get the CPU topology.
68  SystemTopology sysTopo;
69  Thread::getSystemTopology(sysTopo);
70 
71  // Get the max efficiency.
72  uint32_t numEfficiencyClasses = 0;
73  for (size_t iGroup = 0; iGroup < sysTopo.groupInfoElementCount; ++iGroup)
74  {
75  ProcessorGroupInfo const& groupInfo = sysTopo.pGroupInfoArray[iGroup];
76 
77  // Loop through each physical core.
78  for (size_t iCore = 0; iCore < groupInfo.coreInfoElementCount; ++iCore)
79  {
80  numEfficiencyClasses = gtsMax(numEfficiencyClasses, (uint32_t)groupInfo.pCoreInfoArray[iCore].efficiencyClass + 1);
81  }
82  }
83 
84  std::vector<WorkerPool*> workerPoolsByEfficiency(numEfficiencyClasses);
85  std::vector<MicroScheduler*> microSchedulersByEfficiency(numEfficiencyClasses);
86  std::vector<ComputeResource*> computeResourcesByEfficiency(numEfficiencyClasses);
87 
88  // Loop through each processor group.
89  for (size_t iGroup = 0; iGroup < sysTopo.groupInfoElementCount; ++iGroup)
90  {
91  ProcessorGroupInfo const& groupInfo = sysTopo.pGroupInfoArray[iGroup];
92 
93  std::vector<size_t> numThreadsByEfficiencyClass(numEfficiencyClasses);
94  std::vector<AffinitySet> affinityByEfficiencyClass(numEfficiencyClasses);
95 
96  // Loop through each physical core.
97  for (size_t iCore = 0; iCore < groupInfo.coreInfoElementCount; ++iCore)
98  {
99  CpuCoreInfo const& coreInfo = groupInfo.pCoreInfoArray[iCore];
100 
101  // Loop through each thread.
102  for (size_t iThead = 0; iThead < coreInfo.hardwareThreadIdCount; ++iThead)
103  {
104  numThreadsByEfficiencyClass[coreInfo.efficiencyClass]++;
105  affinityByEfficiencyClass[coreInfo.efficiencyClass].set(coreInfo.pHardwareThreadIds[iThead]);
106  }
107  }
108 
109  //
110  // Create the WorkerPool for each efficiency class.
111 
112  for (size_t ii = 0; ii < numEfficiencyClasses; ++ii)
113  {
114  WorkerPoolDesc workerPoolDesc;
115 
116  if (iGroup == 0 && ii == numEfficiencyClasses)
117  {
118  // The Mater thread IS part of this class.
119 
120  // Affinitize the Master thread here since the Worker Pool does not own it.
121  AffinitySet masterAffinity;
122  masterAffinity.set(sysTopo.pGroupInfoArray[0].pCoreInfoArray[0].pHardwareThreadIds[0]);
123  gts::ThisThread::setAffinity(0, masterAffinity);
124  }
125  else
126  {
127  // The Mater thread IS NOT part of this node.
128 
129  // Add filler desc for the master so that Worker threads
130  // are created for all the threads in the class.
131  workerPoolDesc.workerDescs.push_back(WorkerThreadDesc());
132  }
133 
134  size_t numThreads = numThreadsByEfficiencyClass[ii];
135  for (size_t iWorker = 0; iWorker < numThreads; ++iWorker)
136  {
137  WorkerThreadDesc workerDesc;
138  workerDesc.affinity = { iGroup, affinityByEfficiencyClass[ii] };
139  workerPoolDesc.workerDescs.push_back(workerDesc);
140  }
141 
142  WorkerPool* pWorkerPool = alignedNew<WorkerPool, GTS_NO_SHARING_CACHE_LINE_SIZE>();
143  bool result = pWorkerPool->initialize(workerPoolDesc);
144  GTS_ASSERT(result);
145  workerPoolsByEfficiency[ii] = pWorkerPool;
146  }
147  }
148 
149  // Create the MicroSchedulers
150  for (size_t ii = 0; ii < numEfficiencyClasses; ++ii)
151  {
152  auto& vec = workerPoolsByEfficiency[ii];
153 
154  MicroScheduler* pMicroScheduler = alignedNew<MicroScheduler, GTS_NO_SHARING_CACHE_LINE_SIZE>();
155  pMicroScheduler->initialize(workerPoolsByEfficiency[ii]);
156  microSchedulersByEfficiency[ii] = pMicroScheduler;
157 
158  computeResourcesByEfficiency[ii] = alignedNew<MicroScheduler_ComputeResource, GTS_NO_SHARING_CACHE_LINE_SIZE>(pMicroScheduler, 0);
159 
160  // Assumes each efficiency class is ii-times slower than computeResourcesByEfficiency[0];
161  computeResourcesByEfficiency[ii]->setExecutionNormalizationFactor(ii + 1.0);
162  }
163 
164  // Link such that higher efficiency schedulers victimize lower efficiency schedulers.
165  for (int ii = numEfficiencyClasses - 1; ii > 0; --ii)
166  {
167  auto pHighSched = microSchedulersByEfficiency[ii];
168 
169  for (int jj = ii - 1; jj >= 0; --jj)
170  {
171  pHighSched->addExternalVictim(microSchedulersByEfficiency[jj]);
172  }
173  }
174 
175  //
176  // Second, we create a CriticallyAware_MacroScheduler and map the
177  // MicroScheduler_ComputeResources to it.
178 
179  MacroScheduler* pMacroScheduler = new CriticallyAware_MacroScheduler;
180 
181  MacroSchedulerDesc generalizedDagSchedulerDesc;
182  for (uint32_t ii = 0; ii < numEfficiencyClasses; ++ii)
183  {
184  generalizedDagSchedulerDesc.computeResources.push_back(computeResourcesByEfficiency[ii]);
185  }
186 
187  pMacroScheduler->init(generalizedDagSchedulerDesc);
188 
189  //
190  // For this example let's make a random DAG.
191 
192  const uint32_t NUM_RANKS = 10;
193  const uint32_t MIN_NODES_PER_RANK = 2;
194  const uint32_t MAX_NODES_PER_RANK = 3;
195  const uint32_t CHANCE_OF_INCOMING_END = 0;
196 
197  Vector<Node*> nodes;
198 
199  DagUtils::generateRandomDag(pMacroScheduler, 1, NUM_RANKS, MIN_NODES_PER_RANK, MAX_NODES_PER_RANK, CHANCE_OF_INCOMING_END, nodes);
200 
201  for (uint32_t ii = 0; ii < nodes.size(); ++ii)
202  {
203  nodes[ii]->setName("%d\n", ii);
204  nodes[ii]->addWorkload<MicroSchedulerLambda_Workload>([ii](WorkloadContext const&) { printf("%d\n", ii); });
205  }
206 
207  Schedule* pSchedule = pMacroScheduler->buildSchedule(nodes.front(), nodes.back());
208  pMacroScheduler->executeSchedule(pSchedule, computeResourcesByEfficiency[0]->id());
209  pMacroScheduler->freeSchedule(pSchedule);
210 
211  delete pMacroScheduler;
212 
213  for (size_t ii = 0; ii < numEfficiencyClasses; ++ii)
214  {
215  alignedDelete(computeResourcesByEfficiency[ii]);
216  alignedDelete(microSchedulersByEfficiency[ii]);
217  alignedDelete(workerPoolsByEfficiency[ii]);
218  }
219 
220  printf("SUCCESS!\n\n");
221 }
