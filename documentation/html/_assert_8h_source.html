58 GTS_INLINE void _assert(
59  bool expression,
60  const char* exp,
61  const char* file,
62  int line)
63 {
64  if (!expression)
65  {
66  assertHook hook = getAssertHook();
67  if (hook)
68  {
69  constexpr size_t BUFF_SIZE = 2048;
70  static char LOG_BUFFER[BUFF_SIZE];
71  #ifdef GTS_MSVC
72  sprintf_s(LOG_BUFFER, "%s, FILE: %s file (%d)", exp, file, line);
73  #else
74  snprintf(LOG_BUFFER, BUFF_SIZE, "%s, FILE: %s file (%d)", exp, file, line);
75  #endif
76  hook(LOG_BUFFER);
77  }
78  else
79  {
80  printf("MISSING ASSERT HOOK!\n");
81  }
82 
83  GTS_DEBUG_BREAK();
84  }
85 }
