169  template<typename TSplitter>
170  GTS_INLINE void split(split_result& result, TSplitter const& splitter)
171  {
172  result.size = MAX_SPLITS;
173  result.ranges[0] = *this;
174  KdRange3d& r = result.ranges[0];
175 
176  // If xRange is relatively smaller than yRange,
177  if (m_xRange.size() * m_yRange.minSize() < m_yRange.size() * m_xRange.minSize())
178  {
179  // If yRange is relatively smaller than zRange,
180  if (m_yRange.size() * m_zRange.minSize() < m_zRange.size() * m_yRange.minSize())
181  {
182  // Split zRange.
183  r.m_zRange.begin() = z_range_type::splitHelper(m_zRange, splitter);
184  r.m_zRange.end() = m_zRange.end();
185  m_zRange.end() = r.m_zRange.begin();
186  }
187  else
188  {
189  // Otherwise, split yRange.
190  r.m_yRange.begin() = y_range_type::splitHelper(m_yRange, splitter);
191  r.m_yRange.end() = m_yRange.end();
192  m_yRange.end() = r.m_yRange.begin();
193  }
194  }
195  // Otherwise xRange is relatively bigger than yRange, so
196  else
197  {
198  // If xRange is relatively smaller than zRange,
199  if (m_xRange.size()*m_zRange.minSize() < m_zRange.size()*m_xRange.minSize())
200  {
201  // Split zRange.
202  r.m_zRange.begin() = z_range_type::splitHelper(m_zRange, splitter);
203  r.m_zRange.end() = m_zRange.end();
204  m_zRange.end() = r.m_zRange.begin();
205  }
206  else
207  {
208  // Otherwise, split xRange.
209  r.m_xRange.begin() = x_range_type::splitHelper(m_xRange, splitter);
210  r.m_xRange.end() = m_xRange.end();
211  m_xRange.end() = r.m_xRange.begin();
212  }
213  }
214  }
